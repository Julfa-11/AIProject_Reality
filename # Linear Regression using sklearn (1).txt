# Linear Regression with Plot for Pizza Price Prediction

import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

# Training data
X = np.array([[8], [10], [12]])   # Diameters (in inches)
y = np.array([10, 13, 16])        # Prices (in dollars)

# Create and train the model
model = LinearRegression()
model.fit(X, y)

# Predict price for 20-inch pizza
X_pred = np.array([[20]])
y_pred = model.predict(X_pred)

# Print results
print(f"Predicted price for a 20-inch pizza: ${y_pred[0]:.2f}")

# Create a smooth line for the regression
X_line = np.linspace(6, 22, 100).reshape(-1, 1)
y_line = model.predict(X_line)

# Plot the data and regression line
plt.scatter(X, y, color='blue', label='Training Data (Actual)')
plt.plot(X_line, y_line, color='red', label='Regression Line', linewidth=2)
plt.scatter(X_pred, y_pred, color='green', s=100, label=f'Predicted (20-inch: ${y_pred[0]:.2f})')

# Labels and title
plt.title("Pizza Price Prediction using Linear Regression")
plt.xlabel("Diameter (inches)")
plt.ylabel("Price ($)")
plt.legend()
plt.grid(True)
plt.show()
